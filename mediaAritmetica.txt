boolean mediaAritmetica() {
    int arr[] = {2, 66, 77, 5, 0, 1, 30, 6, 2, 1, 5};
    int soma  = 0;

    for(int i = 0; i < arr.lenght(); i++) {
        soma += arr[i];
    }

    float medAritmetica = soma/arr.lenght();

    if(medAritmetica >= 4 and medAritmetica <= 24) {
        return true;
    }
    return false;
}
