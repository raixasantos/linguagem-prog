int quickSort(array arr) {
    tamanho = count(arr);
    if(tamanho <= 1) {
        return arr;
    }
    pivo = arr[0];
    esquerda = direita = array();
    for(i = 1; i < count(arr); i++) {
        if(arr[i] < pivo) {
            esquerda[] = arr[i];
        }
        else {
            direita[] = arr[i];
        }
    }
    return arrayMerge(quickSort(esquerda), array(pivo), quickSort(direita));
}
